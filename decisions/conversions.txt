country_decisions = {
	#
	become_egyptian_dynasty = {
		potential = {
			OR = {
				num_of_cities >= 100
				trigger_if = {
					limit = { is_ai = yes}
					num_of_cities >= 50
				}
			}
			capital_scope = {
				is_in_egypt_trigger = yes
			}
			NOT = { government = egyptian_dynasty }
		}
		allow = {
			trigger_if = {
				limit = {
					is_subject = yes
				}
				is_subject = no
			}
			religion = egyptian_religion
		}
		effect = {
			found_egyptian_dynasty_effect = yes
		}
		
		ai_will_do = {
			factor = 1
		}
	}
	become_tribal_chiefdom = {
		potential = {
			num_of_cities >= 1
			OR = { 
				government = settled_tribal_chiefdom
				government = tribal_kingdom
			}
		}
		allow = {
			centralization < -25
			capital_scope = {
				custom_tooltip = {
					text = pops_to_decivilize_pop
					num_of_tribesmen > num_of_citizen
					num_of_tribesmen > num_of_freemen
				}
			}
			NOT = {
				any_character = {
					is_clan_chief = yes
					loyalty < 33
				}
			}
		}
		effect = {
			add_political_influence = influence_large
			if = {
				limit = {
					government = settled_tribal_chiefdom
				}
				add_stability = -10
			}
			else = {
				add_stability = -30
			}
			change_government = tribal_chiefdom
			switch_government_type_event_clearup_effect = yes
		}
		
		ai_will_do = {
			factor = 0
		}
	}
	become_settled_tribal_chiefdom = {
		potential = {
			num_of_cities >= 1
			government = tribal_chiefdom
		}
		allow = {
			custom_tooltip = {
				text = cannot_stop_migrating_tooltip 
				NOT = {
					any_unit = {
						num_of_migrants > 0
					}
				}
			}
			centralization < 10
			NOT = {
				any_character = {
					is_clan_chief = yes
					loyalty < 33
				}
			}
		}
		effect = {
			add_political_influence = influence_large
			remove_stability_effect = yes
			change_government = settled_tribal_chiefdom
			switch_government_type_event_clearup_effect = yes
		}
		
		ai_will_do = {
			factor = 0
		}
	}
	
	become_tribal_kingdom = {
		potential = {
			num_of_cities >= 1
			OR = {
				government = tribal_chiefdom
				government = settled_tribal_chiefdom
			}
		}
		allow = {
			custom_tooltip = {
				text = cannot_stop_migrating_tooltip 
				NOT = {
					any_unit = {
						num_of_migrants > 0
					}
				}
			}
			centralization > 25
			NOT = {
				any_character = {
					is_clan_chief = yes
					loyalty < 33
				}
			}
		}
		effect = {
			add_political_influence = influence_large
			remove_stability_effect = yes
			change_government = tribal_kingdom
			switch_government_type_event_clearup_effect = yes
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				war = yes
			}
		}
	}
	
	become_despotic_monarchy_tribal = {
		potential = {
			num_of_cities >= 1
			government = tribal_kingdom
		}
		allow = {
			stability >= 40
			centralization >= 80
			capital_scope = { civilization_value >= 40 }
			has_law = absolute_authority
			NOT = {
				has_law = increased_council_powers
			}
			NOT = {
				any_character = {
					is_clan_chief = yes
					loyalty < 33
				}
			}
		}
		effect = {
			add_stability = -10
			change_government = despotic_monarchy
			switch_government_type_event_clearup_effect = yes
			add_country_modifier = {
				name = civilized_tribe
				duration = 5475
			}
			capital_scope = {
				add_civilization_value = 5
			}
			add_2_free_province_investments = yes
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				war = yes
			}
		}
	}
	become_trade_state_monarchy = {
		potential = {
			num_of_cities >= 1
			government = despotic_monarchy
		}
		allow = {
			stability >= 40
			num_of_cities >= 120
			OR = {
				has_country_modifier = bureaucracy_3
				has_country_modifier = bureaucracy_4
			}
		}
		effect = {
			add_stability = -10
			change_government = trade_state_monarchy
			switch_government_type_event_clearup_effect = yes
			add_2_free_province_investments = yes
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				war = yes
			}
		}
	}
	become_religious_state_monarchy = {
		potential = {
			num_of_cities >= 1
			government = despotic_monarchy
		}
		allow = {
			stability >= 40
			num_of_cities >= 120
			OR = {
				has_country_modifier = bureaucracy_3
				has_country_modifier = bureaucracy_4
			}
		}
		effect = {
			add_stability = -10
			change_government = religious_state_monarchy
			switch_government_type_event_clearup_effect = yes
			add_2_free_province_investments = yes
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				war = yes
			}
		}
	}
	become_martial_state_monarchy = {
		potential = {
			num_of_cities >= 1
			government = despotic_monarchy
			OR = {
				has_country_modifier = bureaucracy_3
				has_country_modifier = bureaucracy_4
			}
		}
		allow = {
			stability >= 40
			num_of_cities >= 120
		}
		effect = {
			add_stability = -10
			change_government = martial_state_monarchy
			switch_government_type_event_clearup_effect = yes
			add_2_free_province_investments = yes
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				war = yes
			}
		}
	}
}
