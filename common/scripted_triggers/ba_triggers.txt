trigger_overwrite = {
	$TRIGGER$ >= $VALUE$
}
has_city_trigger = {
	has_city_status = yes
}
papyrus_trigger = {
	OR = {
		is_in_region = middle_egypt_region
		is_in_region = lower_upper_egypt_region
		is_in_region = memphis_region
		is_in_region = eastern_delta_region
		is_in_region = western_delta_region
		is_in_region = upper_upper_egypt_region
		is_in_region = nubia_region
	}
	NOR = {
		terrain = desert
		terrain = desert_hills
		terrain = desert_valley
	}
}

		is_in_region = west_kizzuwatna_region
		is_in_region = east_kizzuwatna_region
	}
}
